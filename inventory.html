const SS_ID = '1MtyzPb0uHlTVpS18ib5k8SNibD02w95iSoHMofEK4AE'; const ALL_TABS = ['overview','JAN','FEB','MAR','APR','MAY','JUN','JUL','AUG','SEP','OCT','NOV','DEC']; function doGet(e) { try { const tab = e.parameter && e.parameter.tab; const ss = SpreadsheetApp.openById(SS_ID); if (tab) { return jsonOut({ ok:true, tab:tab, rows: readTab(ss, tab) }); } const all = {}; ALL_TABS.forEach(t => { all[t] = readTab(ss, t); }); return jsonOut({ ok:true, data:all }); } catch(err) { return jsonOut({ ok:false, error: err.toString() }); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SS_ID); if (body.action === 'save') { writeTab(ss, body.tab, body.rows); return jsonOut({ ok:true }); } return jsonOut({ ok:false, error:'unknown action' }); } catch(err) { return jsonOut({ ok:false, error: err.toString() }); } } function readTab(ss, tabName) { const ws = ss.getSheetByName(tabName); if (!ws || ws.getLastRow() < 2) return []; const headers = ws.getRange(1,1,1,ws.getLastColumn()).getValues()[0].map(h=>h.toString()); const data = ws.getRange(2,1,ws.getLastRow()-1,ws.getLastColumn()).getValues(); return data .filter(r => r.some(c => c !== '')) .map(r => { const obj = {}; headers.forEach((h,i) => { obj[h] = r[i] !== undefined ? String(r[i]) : ''; }); return obj; }); } function writeTab(ss, tabName, rows) { let ws = ss.getSheetByName(tabName); if (!ws) ws = ss.insertSheet(tabName); ws.clearContents(); if (!rows || rows.length === 0) return; const headers = Object.keys(rows[0]); const vals = [headers, ...rows.map(r => headers.map(h => r[h] ?? ''))]; ws.getRange(1,1,vals.length,headers.length).setValues(vals); } function jsonOut(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
